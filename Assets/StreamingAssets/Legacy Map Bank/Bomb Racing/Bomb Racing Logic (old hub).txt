OnTitanDie("t","player0")
{

  VariableBool.Set("playerRegistered", false);
  VariableInt.Set("i",0);
  While(Int.LessThanOrEqual(VariableInt("i"),VariableInt("totalPlayers")))
  {
    VariableString.Set("player","p");
    VariableString.Append("player", VariableInt("i").ConvertToString());

    VariablePlayer.Set("p", VariablePlayer(VariableString("player")));

    If(String.Equals(VariablePlayer("p").GetName(),VariablePlayer("player0").GetName()))
    {
     VariableBool.Set("playerRegistered", true); 
     VariableInt.Set("i", VariableInt("totalPlayers"));
    }

    VariableInt.Add("i",1);
  }

  If (Bool.Equals(VariableBool("playerRegistered"), false))
  {
    VariableString.Set("player","p");
    VariableString.Append("player", VariableInt("totalPlayers").ConvertToString());

    VariablePlayer.Set(VariableString("player"), VariablePlayer("player0"));

    VariableInt.Add("totalPlayers", 1);

OnUpdate()
{
  VariableInt.Add("timer",1);
  VariableInt.Set("i",0);
  While(Int.LessThanOrEqual(VariableInt("i"),VariableInt("totalPlayers")))
  {
    VariableString.Set("player","p");
    VariableString.Append("player", VariableInt("i").ConvertToString());

    VariablePlayer.Set("p", VariablePlayer(VariableString("player")));

    If (Integer.Equals(VariablePlayer("p").GetTeam(), 2))
    {

      If (Boolean.Equals(VariablePlayer("p").GetIsAlive(), true))
      {

        If (Float.GreaterThan(VariablePlayer("p").GetPositionY(), -9800.0))
        {
          If (Bool.Equals(VariablePlayer("p").GetCustomBool(), true))
          {
            VariableString.Set("playerX", VariableString("player"));
            VariableString.Set("playerY", VariableString("player"));
            VariableString.Set("playerZ", VariableString("player"));

            VariableString.Append("playerX", "X");
            VariableString.Append("playerY", "Y");
            VariableString.Append("playerZ", "Z");

            VariableFloat.Set(VariableString("playerX"), VariablePlayer("p").GetPositionX());
            VariableFloat.Set(VariableString("playerY"), VariablePlayer("p").GetPositionY());
            VariableFloat.Set(VariableString("playerZ"), VariablePlayer("p").GetPositionZ());
          }

          If (Bool.Equals(VariablePlayer("p").GetCustomBool(), false))
          {
            Player.MovePlayer(VariablePlayer("p"), VariableFloat("magentaSpawnX"), VariableFloat("magentaSpawnY"), VariableFloat("magentaSpawnZ"));
            Player.SetCustomBool(VariablePlayer("p"), true);
          }
        }

        If (Float.LessThanOrEqualTo(VariablePlayer("p").GetPositionY(), -9800.0))
        {

          If (Float.LessThanOrEqualTo(VariablePlayer("p").GetPositionZ(), -5.0))
          {
            VariableString.Set("playerX", VariableString("player"));
            VariableString.Set("playerY", VariableString("player"));
            VariableString.Set("playerZ", VariableString("player"));

            VariableString.Append("playerX", "X");
            VariableString.Append("playerY", "Y");
            VariableString.Append("playerZ", "Z");

            Player.KillPlayer(VariablePlayer("p"), "Moving_Player");
            Player.SpawnPlayerAt(VariablePlayer("p"), VariableFloat(VariableString("playerX")), VariableFloat(VariableString("playerY")), VariableFloat(VariableString("playerZ")));
            Player.SetCustomBool(VariablePlayer("p"), true);
          }
        }
      }
      If (Boolean.Equals(VariablePlayer("p").GetIsAlive(), false))
      {
        Player.SetCustomBool(VariablePlayer("p"), false);
        Player.SpawnPlayerAt(VariablePlayer("p"), VariableFloat("magentaSpawnX"), VariableFloat("magentaSpawnY"), VariableFloat("magentaSpawnZ"));
      }
    }

    VariableInt.Add("i",1);
  }

  If (Integer.Equals(VariableInt("timer"), 2))
  {
    If (Integer.GreaterThan(VariableInt("numberOfTitans"), 0))
    {
      Titan.SpawnTitanAt(0,3.0,0,1,-52.5,-9984.0,67.5);
    }

    If (Integer.GreaterThan(VariableInt("numberOfTitans"), 1))
    {
      Titan.SpawnTitanAt(0,3.0,0,1,-52.5,-9984.0,127.5);
    }

    If (Integer.GreaterThan(VariableInt("numberOfTitans"), 2))
    {
      Titan.SpawnTitanAt(0,3.0,0,1,-52.5,-9984.0,187.5);
    }

    If (Integer.GreaterThan(VariableInt("numberOfTitans"), 3))
    {
      Titan.SpawnTitanAt(0,3.0,0,1,-52.5,-9984.0,247.5);
    }

    If (Integer.GreaterThan(VariableInt("numberOfTitans"), 4))
    {
      Titan.SpawnTitanAt(0,3.0,0,1,-52.5,-9984.0,307.5);
    }
